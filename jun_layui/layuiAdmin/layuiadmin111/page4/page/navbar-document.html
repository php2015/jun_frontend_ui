LayuiAdmin 内置Navbar组件:根据Layui的导航结构渲染的,以快速搭建导航菜单为目的

快速使用(渲染顶部导航栏) 
layui.config({
    base:'js/'
}).extend({
    'navbar' : 'navbar'
});
layui.use(['navbar'],function(){
    var navbar = layui.navbar;
    navbar.render({
        data: [{
            icon : 'fa-envelope-o',
            badge : 'layui-bg-blue',
            badgeTitle: 9
        },{
            icon : 'fa-bell-o',
            badgeTitle: 6
        },{
            title : '管理人员',
            badgeTitle: 6,
            children : [{
                title : '密码修改'
            },{
                title : '退出'
            }]
        },{
            title : '主题设置',
            class : 'layui-theme',
            children : [{
                title : '默认主题',
                layTheme : 'default'
            },{
                title : '浅蓝主题',
                layTheme : 'blue'
            },{
                title : '墨绿主题',
                layTheme : 'molv'
            },{
                title : '藏青主题',
                layTheme : 'cyan'
            },{
                title : '黑色主题',
                layTheme : 'black'
            }]
        }]
    });
});
快速使用(渲染侧边导航栏) 
layui.config({
    base:'js/'
}).extend({
    'navbar' : 'navbar'
});
layui.use(['navbar'],function(){
    var navbar = layui.navbar;
    navbar.render({
        filter : 'side',
        data : [{
            title : '组件文档',
            icon : 'fa-bars',
            children : [{
                title : '主题',
                icon : 'fa-bars',
                layId : 'theme',
                layUrl : 'page/theme-document.html'
            },{
                title : '导航',
                icon : 'fa-bars',
                layId : 'navbar',
                layUrl : 'page/navbar-document.html'
            }]
        },{
            title : '元市场',
            icon : 'fa-bars',
            layId : 'test',
            layUrl : 'page/test.html'
        }]
    });
});
导航动态操作 
layui.config({
    base:'js/'
}).extend({
    'navbar' : 'navbar'
});
layui.use(['navbar'],function(){
    var navbar = layui.navbar;
    navbar.init();
});
navbar.render(options)
参数 默认值
elem-绑定元素(建议不填写) .layui-nav
filter-渲染对象 navbar
data-渲染数据 [] 
[{
    title : '组件文档',
    icon : 'fa-bars',
    children : [{
        title : '主题',
        icon : 'fa-bars',
        layId : 'theme',
        layUrl : 'page/theme-document.html'
    },{
        title : '导航',
        icon : 'fa-bars',
        layId : 'navbar',
        layUrl : 'page/navbar-document.html'
    }]
},{
    title : '元市场',
    icon : 'fa-bars',
    layId : 'test',
}]
navbar.init(options)
参数 默认值
navFilter-(监听点击的导航值) nav(side)
filter-(选项卡添加的选项卡值) content
reload-(刷新页面绑定的元素) .layui-reload
closeAllTab-(关闭所有选项卡绑定的元素) .layui-close-alltab
closeOtherTab-(关闭其他选项卡绑定的元素) .layui-close-othertab
